# git branches

parse_git_branch() {
    git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/<\1>/'
}

PS1="\${debian_chroot:+($debian_chroot)}\[\e[0;32m\]\u\[\e[m\]:\[\e[1;34m\]\w\[\e[m\]\[\e[1;32m\]\$(parse_git_branch)\[\e[m\]\$ "
